Algoritmo_Calculo_de_salario_liquido
Var salario_bruto, percentual_inss, desconto_inss, salario_liquido Real
Início
Escreva("Digite o salário bruto: ")
    Leia(salario_bruto)
Escreva("Digite o percentual de desconto do INSS: ")
    Leia(percentual_inss)
 desconto_inss ← salario_bruto * percentual_inss / 100
salario_liquido ← salario_bruto - desconto_inss
Escreva("O salário líquido é: ", salario_liquido)
Fim
